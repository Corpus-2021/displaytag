22  package org.displaytag.export;
23  
24  import java.io.IOException;
25  import java.io.OutputStream;
26  
27  import com.itextpdf.text.Document;
28  import com.itextpdf.text.DocumentException;
29  import com.itextpdf.text.pdf.BaseFont;
30  import com.itextpdf.text.pdf.PdfContentByte;
31  import com.itextpdf.text.pdf.PdfPTable;
32  import com.itextpdf.text.pdf.PdfPageEventHelper;
33  import com.itextpdf.text.pdf.PdfTemplate;
34  import com.itextpdf.text.pdf.PdfWriter;
35  
36  
37  /**
38   * PDF exporter using iText.
39   * @author Jorge L. Barroso
40   * @version $Revision$ ($Author$)
41   */
42  public class DefaultPdfExportView extends DefaultItextExportView
43  {
44  
45      /**
46       * @see org.displaytag.export.BaseExportView#getMimeType()
47       * @return "application/pdf"
48       */
49      @Override
50      public String getMimeType()
51      {
52          return "application/pdf"; //$NON-NLS-1$
53      }
54  
55      /**
56       * Initializes the PDF writer this export view uses to write the table document.
57       * @param document The iText document to be written.
58       * @param out The output stream to which the document is written.
59       * @throws DocumentException If something goes wrong during initialization.
60       */
61      @Override
62      protected void initItextWriter(Document document, OutputStream out) throws DocumentException
63      {
64          PdfWriter.getInstance(document, out).setPageEvent(new PageNumber());
65      }
66  
67      /**
68       * Prints a page number at the bottom of each page. Based on
69       * http://itextdocs.lowagie.com/examples/com/lowagie/examples/directcontent/pageevents/PageNumbersWatermark.java
70       * @author Jorge L. Barroso
71       * @version $Revision$ ($Author$)
72       */
73      private static class PageNumber extends PdfPageEventHelper
74      {
75  
76          /**
77           * @see com.lowagie.text.pdf.PdfPageEventHelper#onEndPage(com.lowagie.text.pdf.PdfWriter,
78           * com.lowagie.text.Document)
79           */
80          @Override
81          public void onEndPage(PdfWriter writer, Document document)
82          {
83              /** The headertable. */
84              PdfPTable table = new PdfPTable(2);
85              /** A template that will hold the total number of pages. */
86              PdfTemplate tpl = writer.getDirectContent().createTemplate(100, 100);
87              /** The font that will be used. */
88              BaseFont helv = null;
89              try
90              {
91                  helv = BaseFont.createFont("Helvetica", BaseFont.WINANSI, false);
92              }
93              catch (DocumentException e)
94              {
95              }
96              catch (IOException e)
97              {
98              }
99              PdfContentByte cb = writer.getDirectContent();
100             // cb.saveState();
101             // write the headertable
102             table.setTotalWidth(document.right() - document.left());
103             table.writeSelectedRows(0, -1, document.left(), document.getPageSize().getHeight() - 50, cb);
104             // compose the footer
105             String text = "Page " + writer.getPageNumber();
106             float textSize = helv.getWidthPoint(text, 12);
107             float textBase = document.bottom() - 20;
108             cb.beginText();
109             cb.setFontAndSize(helv, 12);
110             float adjust = helv.getWidthPoint("0", 12);
111             cb.setTextMatrix(document.right() - textSize - adjust, textBase);
112             cb.showText(text);
113             cb.endText();
114             cb.addTemplate(tpl, document.right() - adjust, textBase);
115             // cb.saveState();
116         }
117     }
118 }
