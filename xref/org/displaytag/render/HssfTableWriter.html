148     /**
149      * @see org.displaytag.render.TableWriterTemplate#writeCaption(org.displaytag.model.TableModel)
150      */
151     @Override
152     protected void writeCaption(TableModel model) throws Exception
153     {
154         HSSFCellStyle style = this.wb.createCellStyle();
155         HSSFFont bold = this.wb.createFont();
156         bold.setBoldweight(Font.BOLDWEIGHT_BOLD);
157         bold.setFontHeightInPoints((short) 14);
158         style.setFont(bold);
159         style.setAlignment(CellStyle.ALIGN_CENTER);
160 
161         this.colNum = 0;
162         this.currentRow = this.sheet.createRow(this.sheetRowNum++);
163         this.currentCell = this.currentRow.createCell(this.colNum);
164         this.currentCell.setCellStyle(style);
165         String caption = model.getCaption();
166         this.currentCell.setCellValue(new HSSFRichTextString(caption));
167         this.rowSpanTable(model);
168     }

181     /**
182      * @see org.displaytag.render.TableWriterTemplate#writeTableHeader(org.displaytag.model.TableModel)
183      */
184     @Override
185     protected void writeTableHeader(TableModel model) throws Exception
186     {
187         this.currentRow = this.sheet.createRow(this.sheetRowNum++);
188         this.colNum = 0;
189         HSSFCellStyle headerStyle = this.getHeaderFooterStyle();
190         for (HeaderCell headerCell : model.getHeaderCellList())
191         {
192             String columnHeader = headerCell.getTitle();
193             if (columnHeader == null)
194             {
195                 columnHeader = StringUtils.capitalize(headerCell.getBeanPropertyName());
196             }
197 
198             this.writeHeaderFooter(columnHeader, this.currentRow, headerStyle);
199         }
200     }

234     /**
235      * @see org.displaytag.render.TableWriterTemplate#writeColumnValue(Object,org.displaytag.model.Column)
236      */
237     @Override
238     protected void writeColumnValue(Object value, Column column) throws Exception
239     {
240         // is this a detail row for a column that is currently grouped?
241         int myGroup = column.getHeaderCell().getGroup();
242         Object cellValue = column.getValue(this.decorated);
243         if (myGroup > 0)
244         {
245             cellValue = "";
246         }
247         writeCellValue(cellValue);
248     }

250     /**
251      * Override in subclasses to handle local data types.
252      * @param value the value object to write
253      */
254     protected void writeCellValue(Object value)
255     {
256         if (value instanceof Number)
257         {
258             Number num = (Number) value;
259             // Percentage
260             if (value.toString().indexOf("%") > -1)
261             {
262                 this.currentCell.setCellValue(num.doubleValue() / 100);
263                 this.currentCell.setCellStyle(this.utils.getStyle(ExcelUtils.STYLE_PCT));
264             }
265             else if (value instanceof Integer)
266             {
267                 this.currentCell.setCellStyle(this.utils.getStyle(ExcelUtils.STYLE_INTEGER));
268                 this.currentCell.setCellValue(num.intValue());
269             }
270             else
271             {
272                 this.currentCell.setCellValue(num.doubleValue());
273             }
274 
275         }
276         else if (value instanceof Date)
277         {
278             this.currentCell.setCellValue((Date) value);
279             this.currentCell.setCellStyle(this.utils.getStyle(ExcelUtils.STYLE_DATE));
280         }
281         else if (value instanceof Calendar)
282         {
283             Calendar c = (Calendar) value;
284             this.currentCell.setCellValue(c);
285             this.currentCell.setCellStyle(this.utils.getStyle(ExcelUtils.STYLE_DATE));
286         }
287         else if (value == null)
288         {
289             this.currentCell.setCellValue(EMPTY_TEXT);
290         }
291         else
292         {
293             String v = value.toString();
294             if (v.length() > this.utils.getWrapAtLength())
295             {
296                 this.currentCell.getCellStyle().setWrapText(true);
297             }
298             this.currentCell.setCellValue(new HSSFRichTextString(ExcelUtils.escapeColumnValue(value)));
299         }
300 
301     }

307     @Override
308     protected void writeDecoratedRowFinish(TableModel model) throws Exception
309     {
310         TableDecorator decorator = model.getTableDecorator();
311         if (decorator instanceof DecoratesHssf)
312         {
313             DecoratesHssf hdecorator = (DecoratesHssf) decorator;
314             hdecorator.setSheet(this.sheet);

318         this.sheetRowNum++;

353     /**
354      * Is this value numeric? You should probably override this method to handle your locale.
355      * @param rawValue the object value
356      * @return true if numeric
357      */
358     protected boolean isNumber(String rawValue)
359     {
360         if (rawValue == null)
361         {
362             return false;
363         }
364         String rawV = rawValue;
365         if (rawV.indexOf('%') > -1)
366         {
367             rawV = rawV.replace('%', ' ').trim();
368         }
369         if (rawV.indexOf('$') > -1)
370         {
371             rawV = rawV.replace('$', ' ').trim();
372         }
373         if (rawV.indexOf(',') > -1)
374         {
375             rawV = StringUtils.replace(rawV, ",", "");
376         }
377         return NumberUtils.isNumber(rawV.trim());
378     }

393     /**
394      * Obtain the style used to render a header or footer.
395      * @return The style used to render a header or footer.
396      */
397     private HSSFCellStyle getHeaderFooterStyle()
398     {
399         HSSFCellStyle style = this.wb.createCellStyle();
400         // style.setFillPattern(HSSFCellStyle.FINE_DOTS);
401         // style.setFillBackgroundColor(HSSFColor.BLUE_GREY.index);
402         HSSFFont bold = this.wb.createFont();
403         bold.setBoldweight(Font.BOLDWEIGHT_BOLD);
404         // bold.setColor(HSSFColor.WHITE.index);
405         style.setBorderBottom(CellStyle.BORDER_THIN);
406         style.setBottomBorderColor(IndexedColors.BLACK.getIndex());
407 
408         style.setFont(bold);
409         return style;

426     @Override
427     protected void writeSubgroupStart(TableModel model) throws Exception
428     {
429         TableTotaler tt = model.getTotaler();
430         if (tt.howManyGroups == 0)
431         {
432             return;
433         }
434 
435         // for each newly opened subgroup we need to output the opener, in order;
436         // so we need to know somehow which groups are new since we last wrote out openers; how about we track a list of
437         // the
438         // already opened groups, and ask the tt for a list of all known groups?
439 
440         for (int dtColumnNumber : tt.getOpenedColumns())
441         {
442             this.currentGrouping++;
443             writeRowOpener(null);
444             // for each subgroup
445 
446             for (HeaderCell cell : model.getHeaderCellList())
447             {
448                 writeColumnOpener(null);
449                 int thisCellAsDtNumber = asDtColNumber(cell.getColumnNumber());
450                 String columnValue = (thisCellAsDtNumber != dtColumnNumber) ? "" : tt.getGroupingValue(dtColumnNumber);
451                 writeCellValue(columnValue);
452                 writeColumnCloser(null);
453             }
454 
455             writeRowCloser(null);
456             // Have to handle a case where this is a nested subgroup start;
457             // put out the blanks for any column that has already exists
458             // now write the label for the group that is opening
459         }
460     }

478     @Override
479     protected void writeSubgroupStop(TableModel model) throws Exception
480     {
481         TableTotaler tt = model.getTotaler();
482 
483         // for each newly opened subgroup we need to output the opener, in order;
484         // so we need to know somehow which groups are new since we last wrote out openers; how about we track a list of
485         // the
486         // already opened groups, and ask the tt for a list of all known groups?
487 
488         if (tt.howManyGroups == 0)
489         {
490             return;
491         }
492         List<Integer> closedColumns = tt.getClosedColumns();
493         Collections.reverse(closedColumns);
494         for (int columnNumber : closedColumns)
495         {
496             writeRowOpener(null);
497             // for each subgroup
498 
499             for (HeaderCell cell : model.getHeaderCellList())
500             {
501                 writeColumnOpener(null);
502                 Object columnValue;
503                 int cellColumnNumberAsDt = asDtColNumber(cell.getColumnNumber());
504                 if (cellColumnNumberAsDt > columnNumber && cell.isTotaled())
505                 {
506                     columnValue = tt.getTotalForColumn(cell.getColumnNumber(), this.currentGrouping);
507                 }
508                 else if (cellColumnNumberAsDt == columnNumber)
509                 {
510                     columnValue = getTotalLabel(tt.getGroupingValue(columnNumber));
511                 }
512                 else
513                 {
514                     columnValue = null;
515                 }
516                 writeCellValue(columnValue);
517                 writeColumnCloser(null);
518             }
519 
520             writeRowCloser(null);
521             writeGroupExtraInfo(model);
522             this.currentGrouping--;
523         }
524 
525         assert this.currentGrouping > -1;
526         super.writeSubgroupStop(model);
527     }

556     @Override
557     protected void writeTableBodyCloser(TableModel model) throws Exception
558     {
559         // write totals, if there are any
560         boolean hasTotals = false;
561         for (HeaderCell cell : model.getHeaderCellList())
562         {
563             hasTotals = hasTotals || cell.isTotaled();
564         }
565         if (!hasTotals)
566         {
567             return;
568         }
569         TableTotaler tt = model.getTotaler();
570         writeRowOpener(null);
571         for (HeaderCell cell : model.getHeaderCellList())
572         {
573             writeColumnOpener(null);
574             Object columnValue = (cell.isTotaled()) ? tt.getTotalForColumn(cell.getColumnNumber(), 0) : null;
575             writeCellValue(columnValue);
576             CellStyle st = this.utils.getNewCellStyle();
577             st.cloneStyleFrom(this.currentCell.getCellStyle());
578             st.setBorderTop(CellStyle.BORDER_THIN);
579             st.setTopBorderColor(IndexedColors.BLACK.getIndex());
580             this.currentCell.setCellStyle(st);
581             writeColumnCloser(null);
582         }
583         writeRowCloser(null);
584     }
